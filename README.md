# - Installation.

________________________
<br>
$ apt install php git -y <br>
$ wget https://raw.githubusercontent.com/ARSecurityTeam/phpServerHost/main/php-server.sh <br>
$ cd phpServerHost <br>
$ chmod +x php-server.sh <br>
$ ./php-server.sh <br>
